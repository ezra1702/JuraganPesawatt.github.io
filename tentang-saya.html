1.Cara Menentukan Bilangan Prima pada Javascript #asaLogika

class BilPrima{ //Buat sebua kelas yang bernama Bil Prima
result = [ ]; //buat sebuah variabel result dengan isinya sebuah array kosong, untuk menampung habis bagi
constructor(z){
this.z = z;
}
prima(){
for(let x = 2; x <= this.z; x++){ //buat sebuah looping yang mulai dari 2, karena bilangan prima mulainya dari 2
var bilangan = 1;// prima
for(let y = 2; y < x; y++){
if (x % y == 0 ) { // untuk menentukan syarat apakah bilangan yang dimasukan bilanga prima atau bukan
bilangan = -1;//not prima
break;
} }
if (x > 1 && bilangan == 1) {
this.result.push(x) //menambahkan bilangan prima kedalam isi result
}
}
return this.result; //kembalikan/ masukan data kedalam data.prima didalam console.info
}
}
const simpan_primma = new BilPrima(m = prompt("Masukan Angka ?")); //Kemudaian buat sebuah variabel constructur untuk memanggil class BilPrima
document.write(`Bilangan Prima dari ${m} = ${simpan_primma.prima()} Pasti dengan yakin hasilnya pasti sama`);

2.Cara Memakai Filter , Map dan Reduce pada Javascript #nambahPengetahuan

const angka = [0,1,5,6,2,9,10,-2]; const huruf = ['kamu','makan','cabe'] // use filter const newHuruf = huruf.filter( (i) => i == 'cabe'); const newBuah = angka.filter( (x) => { return x <= 20; }) document.getElementById('demo').innerHTML = "Nilai berisi " + newHuruf +' ' + newBuah;
// Map // Kalikan semua angka dengan 2 const newAngka = angka.map(a => a + 2 + ' ' + huruf[0]) document.getElementById('demo').innerHTML = "Nilai berisi " + newAngka;
// reduce // jumlahkan seluruh element pada Array const newAngka = angka.reduce((acumulator, curentValue ) => { return acumulator + curentValue; }); console.log(newAngka);
// method chaining // cari angka > 5 // kalikan anka 3 // jumlahkan const hasil = angka.filter( a => a > 5) .map( a => a * 3) .map( a => a%100) .reduce((acc, cur) => acc + cur); document.getElementById('demo').innerHTML = "Nilai berisi " + hasil;
Yang Ini Mimin masih mempelajari, yang ini memang menurut mimin susah
